class Node(object):
def init(self, data):
self.data = data
self.next = None

    def set_data(self, data):
        self.data = data

    def set_next(self, node):
        self.next = node

    def __repr__(self):
        return self.data

class LinkedList(object):
    def __init__(self):
        self.head = None

    def append_item(self, item):
        """
        添加节点
        :param item:
        :return:
        """
        node = Node(item)
        if self.head == None:
            self.head = node
            return
        h_node = self.head
        while True:
            if h_node.next is None:
                h_node.set_next(node)
                break
            h_node = h_node.next

    def remove_item(self, item):
        """
        移除指定链表节点
        :param item:
        :return:
        """
        h_node = self.head
        if h_node.data == item:
            self.head = h_node.next
        else:
            while (h_node):
                if h_node.next.data == item:
                    h_node.set_next(h_node.next.next)
                h_node = h_node.next


    def remove_by_index(self, index):
        """
        移除对应下标index的节点
        :param index:
        :return:
        """
        h_node = self.head
        if index == 0:
            self.head = h_node.next
        else:
            for _ in range(index-1):
                h_node = h_node.next

            h_node.set_next(h_node.next.next)

    def add_item(self, item, index):
        """
        添加节点到链表的指定位置
        :param item:
        :param index:
        :return:
        """
        h_node = self.head
        a_node = Node(item)

        if index == 0:
            a_node.set_next(h_node)
            self.head = a_node
        else:
            for _ in range(index-1):
                print(1)
                h_node = h_node.next
            a_node.set_next(h_node.next)
            h_node.set_next(a_node)


    def size(self):
        """
        获取链表长度
        :return:
        """
        h_node = self.head
        flag = 0
        while h_node:
            h_node = h_node.next
            flag += 1
        return flag

    def get_item(self, index):
        """
        通过下标获取对应节点的值
        :param index:
        :return:
        """
        h_node = self.head
        if index == 0:
            return h_node.data
        else:
            for _ in range(index):
                h_node = h_node.next
            return h_node.data


    def index(self, item):
        """
        获取指定节点下标
        :param item:
        :return:
        """
        h_node = self.head
        list1 = []
        flag = 0
        while h_node:
            if h_node.data == item:
                list1.append(flag)
            flag += 1
            h_node = h_node.next
        return list1


    def __repr__(self):
        """
        输出魔法方法
        :return:
        """
        h_node = self.head
        list1 = []
        while (h_node):
            list1.append(h_node.data)
            h_node = h_node.next
        print(str(list1))
        print(type(str(list1)))
        return str(list1)

linkedlist表自动化测试
import unittest
from linkedlist import LinkedList

class TestIlinkedIist(unittest.TestCase):

    def test_append_item(self):
        l = LinkedList()
        l.append_item('a')

        num = l.get_item(0)
        self.assertTrue(num == 'a')

        size = l.size()
        self.assertTrue(size == 1)

        l.append_item(['a','b','c'])
        size = l.size()
        self.assertTrue(size == 2)

        size = l.size()
        self.assertTrue(size == 2)

        re = l.__repr__()
        self.assertTrue(re == str(['a', ['a','b','c']]))

        l.append_item(3)
        l.append_item(5)


    def test_remove_item(self):
        ll = LinkedList()
        ll.append_item('a')
        ll.append_item(['a','b','c'])
        ll.append_item(3.5)
        ll.append_item(5)

        rell= ll.__repr__()
        self.assertTrue(rell == str(['a', ['a','b','c'],3.5, 5]))

        ll.remove_item(3.5)
        nu = ll.size()
        self.assertTrue(nu == 3)

        rell = ll.__repr__()
        self.assertTrue(rell == str(['a', ['a', 'b', 'c'], 5]))

        ge = ll.get_item(1)
        self.assertTrue(ge == ['a','b','c'])


if __name__ == '__main__':
    unittest.main()
